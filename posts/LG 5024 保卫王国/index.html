LG 5024 保卫王国
noip树链剖分动态dp倍增动态规划树形dp

解法一

最小权覆盖集 = 全集 - 最大权独立集

动态dp

只要把强制取点,不取点可以分别把点的权值改为正无穷与负无穷

解法二

倍增/树链剖分+树形dp

每次修改的两个点a,b构成一条链

可以链上连接的子树与链中间的点的结果不受修改影响,只有a,b的转移受影响,需要单独处理

那么我们可以把子树的答案都预处理出来,链中间部分可以用倍增或树链剖分维护

这样只需要处理a,b

设f(i,0/1)表示i选/不选,以i为根的子树的最小代价,g(i,0/1)表示整棵树除了以i为根的子树的最小代价,可以通过树形dp求出

F(i,j,0/1,0/1)表示i选/不选,i的2^j级祖先选/不选,i子树到i的i的2^j级祖先子树的最小代价,可以倍增求出

#include<bits/stdc++.h>
typedef long long ll;
const int N=100011;const ll inf=1ll<<60;
int n,m,p[N],cnt,head[N],d[N],fa[20][N];
ll f[N][2],g[N][2],F[N][20][2][2];
struct edge{int to,nxt;}e[N*2];
void add(int x,int y){e[++cnt].to=y,e[cnt].nxt=head[x],head[x]=cnt;}
#define fl(i,x) for(int i=head[x],to;to=e[i].to,i;i=e[i].nxt)
ll min(ll x,ll y){return x<y?x:y;}
void cmin(ll&x,ll y){if(y<x)x=y;}
void dfs1(int x){
    f[x][1]=p[x];d[x]=d[fa[0][x]]+1;
    fl(i,x)if(to!=fa[0][x])
        fa[0][to]=x,dfs1(to),
        f[x][0]+=f[to][1],
        f[x][1]+=min(f[to][0],f[to][1]);
}
void dfs2(int x){
    fl(i,x)if(to!=fa[0][x]){
        g[to][0]=g[x][1]+f[x][1]-min(f[to][0],f[to][1]);
        g[to][1]=min(g[x][0]+f[x][0]-f[to][1],g[to][0]);
        dfs2(to);
    }
}
void swap(int&x,int&y){x^=y^=x^=y;}
ll solve(int x,int a,int y,int b){
    if(d[x]<d[y])swap(x,y),swap(a,b);
    ll nx[2],ny[2],tx[2]={inf,inf},ty[2]={inf,inf};
    tx[a]=f[x][a],ty[b]=f[y][b];
    for(int i=19;~i;--i)
        if(d[fa[i][x]]>=d[y]){
            nx[0]=nx[1]=inf;
            for(int u=0;u<2;++u)
                for(int v=0;v<2;++v)
                    cmin(nx[u],tx[v]+F[x][i][v][u]);
            tx[0]=nx[0],tx[1]=nx[1],x=fa[i][x];
        }
    if(x==y)return tx[b]+g[x][b];
    for(int i=19;~i;--i)
        if(fa[i][x]!=fa[i][y]){
            nx[0]=nx[1]=ny[0]=ny[1]=inf;
            for(int u=0;u<2;++u)
                for(int v=0;v<2;++v)
                    cmin(nx[u],tx[v]+F[x][i][v][u]),
                    cmin(ny[u],ty[v]+F[y][i][v][u]);
            tx[0]=nx[0],tx[1]=nx[1],x=fa[i][x],
            ty[0]=ny[0],ty[1]=ny[1],y=fa[i][y];
        }
    int lca=fa[0][x];
    return min(
        f[lca][0]+g[lca][0]-f[x][1]-f[y][1]+tx[1]+ty[1],
        f[lca][1]+g[lca][1]-min(f[x][0],f[x][1])-min(f[y][0],f[y][1])+min(tx[0],tx[1])+min(ty[0],ty[1])
    );
}
int main(){
    int a,x,b,y;char typ[5];
    scanf("%d%d%s",&n,&m,typ);
    for(int i=1;i<=n;++i)scanf("%d",p+i);
    for(int i=1;i<n;++i)
        scanf("%d%d",&x,&y),add(x,y),add(y,x);
    dfs1(1),dfs2(1);
    for(int i=1;i<=n;++i){
        F[i][0][0][0]=inf;
        F[i][0][0][1]=f[fa[0][i]][1]-min(f[i][0],f[i][1]);
        F[i][0][1][0]=f[fa[0][i]][0]-f[i][1];
        F[i][0][1][1]=f[fa[0][i]][1]-min(f[i][0],f[i][1]);
    }
    for(int k=1;k<=19;++k)
        for(x=1;x<=n;++x){
            y=fa[k-1][x],fa[k][x]=fa[k-1][y];
            for(int u=0;u<2;++u)
                for(int v=0;v<2;++v){
                    F[x][k][u][v]=inf;
                    for(int w=0;w<2;++w)
                        cmin(F[x][k][u][v],F[x][k-1][u][w]+F[y][k-1][w][v]);
                }
        }
    while(m--){
        scanf("%d%d%d%d",&a,&x,&b,&y);
        if(!x&&!y&&(fa[0][a]==b||fa[0][b]==a))puts("-1");
        else printf("%lld\n",solve(a,x,b,y));
    }
}
