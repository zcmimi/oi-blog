LG 2511 [HAOI2008]木棍分割

首先二分出最大一段的最小值,然后dp找出方案数

设f[i][j]表示前j个分成i组的方案数,求出的最小值为x

f[i][j] = \sum(f[i-1][k])(S_j-S_k \le x,k \le j-1)

\Theta(n^3m)当然会TLE

我们发现只要找到最小的k之后[k,j-1]都是合法的

这时我们就可以前缀和优化dp

设sum[i][j] = \sum_{k=0}^j f[i][k]

f[i][j]=sum[i-1][j-1] - sum[i-1][k-1]

可是\Theta(n^2m)还是TLE

我们可以先预处理出第一个S_j-S_k \le x的k

因为S_i具有单调性,所以k不需要每次都重新枚举

还没结束

因为f[i][j]每次更新的时候只需要用到f[i-1][...]这一维,我们可以用滚动数组滚掉数组的一维

但是\Theta(nm+n \log \sum L_i)的复杂度还是觉得很悬

#define N 50011
#define mod 10007
int n,m,a[N];
il int mo(int x){
    if(x>=mod)x-=mod;
    return x;
}
il bool chk(int w){
    int s=0,t=0;
    Fur(i,1,n){
        s+=a[i];
        if(s>w){
            s=a[i],++t;
            if(t>m+1)return 0;
        }
    }
    if(s)++t;
    if(t>m+1)return 0;
    return 1;
}
int f[N],s[N],S[N],pre[N];
int main(){
    fin("in");
    in>>n>>m;
    int l=0,r=0,ans=0;
    Fur(i,1,n)
        in>>a[i],
        l=MAX(l,a[i]),
        S[i]=S[i-1]+a[i];

    r=S[n];
    while(l<=r){
        int mid=(l+r)>>1;
        if(chk(mid))ans=mid,r=mid-1;
        else l=mid+1;
    }
    out<<ans<<" ";

    int k=0,res=(S[n]<=ans);
    Fur(i,1,n){
        while(S[i]-S[k]>ans)++k;
        pre[i]=k;
    }
    Fur(j,1,n)
        f[j]=(S[j]<=ans),
        s[j]=mo(s[j-1]+f[j]);

    Fur(i,2,m+1){
        Fur(j,1,n){
            f[j]=s[j-1];
            if(pre[j]-1>=0)f[j]=mo(f[j]-s[pre[j]-1]+mod);
        }
        Fur(j,1,n)
            s[j]=mo(s[j-1]+f[j]);
        res=mo(res+f[n]);
    }

    out<<res<<ln;
}
